\section{ Green's functions}
  Working in the free theory tells us that $
c_{ {\bf{k}},s }(t) \mbox{         } = \mbox{            }
c_{ {\bf{k}},s} \mbox{            } e^{ - i \epsilon_{{\bf{k}}} \mbox{  } t }$ by using this, we can write $$N_{>}=\sum_{ {\bf{p}} }c_{ {\bf{p}}, < }c^{\dagger}_{ {\bf{p}}, < }=\sum_{\bf{p}}\theta(k_F-|p|)(1-n_{{\bf{p}}} )$$  and, $N^{'}_{>}=\sum_{ {\bf{p}} }c^{\dagger}_{ {\bf{p}}, > }c_{ {\bf{p}}, > }=\sum_{ {\bf{p}} }\theta(|p|-k_F)n_{ {\bf{p}}}$. In addition, the $e^{ \pm i \epsilon_{{\bf{p}}} \mbox{  } t} $ functions cancel as $n_{{\bf{p}}}(t)=c_{{\bf{p}}}^{\dagger}(t)c_{{\bf{p}}}(t)$.

=============================================================================================================
The greater than part of the  ` Green's' function can be written as,
\[
<T  e^{ - \lambda N_{>}(t) } e^{ - \lambda^{'} N^{'}_{>}(t) }  c_{ {\bf{k}},> }(t) c^{\dagger}_{ {\bf{k}},> }(t^{'}) >    \]
\[=\theta(t - t')e^{ - i \epsilon_{{\bf{k}}} \mbox{  } (t-t') } < e^{-\lambda N_>} e^{-\lambda' N'_>} c^\dagger_{{\bf{k}},>} c_{{\bf{k}},>} >\]
\[- \theta(t' - t)e^{ - i \epsilon_{{\bf{k}}} \mbox{  } (t'-t) }  < e^{-\lambda N_>} e^{-\lambda' N'_>} c_{{\bf{k}},>} c^\dagger_{{\bf{k}},>} >\]
=============================================================================================================
The lesser than part of the  ` Green's' function can be written as,
\[
<T  e^{ - \lambda N_{>}(t) } e^{ - \lambda^{'} N^{'}_{>}(t) }  c_{ {\bf{k}},< }(t) c^{\dagger}_{ {\bf{k}},< }(t^{'}) >    \]
\[=\theta(t - t')e^{ - i \epsilon_{{\bf{k}}} \mbox{  } (t-t') } < e^{-\lambda N_>} e^{-\lambda' N'_>} c^\dagger_{{\bf{k}},<} c_{{\bf{k}},<} >\]
\[- \theta(t' - t)e^{ - i \epsilon_{{\bf{k}}} \mbox{  } (t'-t) }  < e^{-\lambda N_>} e^{-\lambda' N'_>} c_{{\bf{k}},<} c^\dagger_{{\bf{k}},<} >\]
=============================================================================================================

\subsection{Derivation of $<e^{ - \lambda N_{>} }e^{-\lambda' N'_>}> \mbox{     }$}
For an operator $\hat{O}$, its trace is given by:
\[
\langle \hat{O} \rangle = \frac{\mathrm{Tr} \left( e^{-\beta (\hat{H} - \mu \sum_{{\bf{p}}}n_{{\bf{p}}})} \hat{O} \right)}{\mathrm{Tr} \left( e^{-\beta (\hat{H} - \mu \sum_{{\bf{p}}}n_{{\bf{p}}})} \right)},
\]
Hence, in our case, $\hat{H}=\sum_{{\bf{p}}}\epsilon_{{\bf{p}}} n_{{\bf{p}}}$, where $< e^{-\lambda N_>} e^{-\lambda' N'_>} n_{{\bf{p}},>}>$ we need to substitute $N_{>}=\sum_{\bf{p}}\theta(k_F-|p|)(1-n_{ p  } )$ and $N^{'}_{>}=\sum_{ {\bf{p}} }c^{\dagger}_{ {\bf{p}}, > }c_{ {\bf{p}}, > }=\sum_{ {\bf{p}} }\theta(|p|-k_F)n_{ {\bf{p}}}$ in the Grand canonical ensemble,

\[ 
\mbox{           }<e^{ - \lambda N_{>} }e^{-\lambda' N'_>}>\mbox{     }  = 
\frac{Tr(e^{ - \sum_p (\lambda  \mbox{  }\theta(k_F-|p|) + n_{{\bf{p}}}  [ \beta  (\epsilon_{{\bf{p}}}-\mu) - \lambda  \mbox{  }\theta(k_F-|p|)\mbox{     } +\lambda'\mbox{  }\theta(|p|-k_F)])})}
{Tr(e^{ - \beta \sum_{p} n_{{\bf{p}}}  (\epsilon_{{\bf{p}}}-\mu) } )}
\]
Since Trace means summing over the diagonal eigenvalue of operators, here $n_{{\bf{k}}}\to \{0,1\}$. 
\[ 
\mbox{           }<e^{ - \lambda N_{>} }e^{-\lambda' N'_>}>\mbox{     }  = \]
\[
=\prod_{{\bf{p}}}
\left(
\frac{e^{ -\lambda  \mbox{  }\theta(k_F-|p|)}
(1 + e^{ -   [ \beta  (\epsilon_{{\bf{p}}}-\mu) - \lambda  \mbox{  }\theta(k_F-|p|)+\lambda'  \mbox{  }\theta(|p|-k_F)]  } )}
{  (1 + e^{ - \beta      (\epsilon_{{\bf{p}}}-\mu) } )} \right)
\]
=============================================================================================================
\[
\mbox{           }<e^{ - \lambda N_{>} }e^{-\lambda' N'_>}> \mbox{     }  =
\prod_{{\bf{p}}}
\left(
\frac{
e^{-n_F(p)\lambda  \mbox{  }} + e^{ -   [ \beta  (\epsilon_{{\bf{p}}}-\mu)+(1-n_F(p))\lambda'  \mbox{  }]  } }
{  (1 + e^{ - \beta      (\epsilon_{{\bf{p}}}-\mu) } )} \right)=\xi(\lambda,\lambda')
\]
