\left(e^{\lambda + \lambda^{'} } \theta (t-t^{'})+\theta (t^{'}-t)\right) }

 \mbox{       } + \mbox{            }

\[
g_{max}\mbox{       } = \mbox{            }

\section{Simplification}

Using a more compact notation,
\begin{eqnarray}
& G_{ {\bf{k}},> }(\lambda, \lambda^{'}; t,t^{'}) \mbox{       } \equiv \mbox{       } 
<T  e^{ - \lambda N_{>}(t) } e^{ - \lambda^{'} N^{'}_{>}(t) }  c_{ {\bf{k}},> }(t) c^{\dagger}_{ {\bf{k}},> }(t^{'}) > ; \nonumber\\&  
\mbox{  }G_{ {\bf{k}},< }(\lambda, \lambda^{'}; t,t^{'}) \mbox{       } \equiv \mbox{       }<T  e^{ - \lambda N_{>}(t) } e^{ - \lambda^{'} N^{'}_{>}(t) }  c_{ {\bf{k}}, < }(t) c^{\dagger}_{ {\bf{k}},<  }(t^{'}) >
\end{eqnarray}
and,
\begin{eqnarray}
  F_{\bf k <}({\bf q}; t_1) &= \langle T e^{-\lambda N_>(t) -\lambda' N'_>(t)} \mbox{ } c_{\bf k+q >}(t) \mbox{ } a^\dagger_{{\bf k} + \frac{{\bf q}} {2} }({\bf q}, t_1 )\mbox{ }  c^\dagger_{\bf k<}(t')  \rangle\nonumber\\
  F_{\bf k >}({\bf q}; t_1) &= \langle T e^{-\lambda N_>(t) -\lambda' N'_>(t)} \mbox{ } c_{\bf k-q <}(t) \mbox{ }a_{{\bf k} - \frac{{\bf q}} {2} }({\bf q}, t_1 ) \mbox{ }  c^\dagger_{\bf k>}(t')  \rangle
\end{eqnarray}
We get,
 \[
f_{max}\mbox{       } = \mbox{            }
 \frac{G_{ {\bf{k}}-{\bf{q}},> }(\lambda, \lambda^{'}; t,t^{'}) \mbox{       } \mbox{             }  n_F({\bf{k}}) e^{\frac{i  t^{'} (2 {\bf{k.q}}-q^2)}{2 m}} \mbox{        }
\left(e^{\lambda + \lambda^{'} } \theta (t-t^{'})+\theta (t^{'}-t)\right) }
{e^{\frac{\beta (2 {\bf{k.q}}-q^2)}{2 m}}-e^{\lambda + \lambda^{'} }}
\]
\[
 \mbox{       } + \mbox{            }
 \frac{  G_{ {\bf{k}},< }(\lambda, \lambda^{'}; t,t^{'}) \mbox{       } 
\mbox{             }
( n_F( {\bf{k}}-{\bf{q}} )-1) e^{\frac{i  t (2 {\bf{k.q}} -q^2 )}{2 m}+\lambda + \lambda^{'} }}
{e^{\frac{\beta (2 {\bf{k.q}}-q^2)}{2 m}}-e^{\lambda + \lambda^{'} }}
\]
and,
\[
g_{max}\mbox{       } = \mbox{            }
 \frac{ e^{-\frac{i q^2 t^{'} }{2 m}}\mbox{         }G_{ {\bf{k}}-{\bf{q}},< }(\lambda, \lambda^{'}; t,t^{'}) \mbox{       }  \mbox{             }( n_F({\bf{k}})-1) \mbox{            }
\left(-e^{\frac{i {\bf{k.q}} }{m} t^{'} }\right) \left(e^{\lambda + \lambda^{'} }
\mbox{              }  \theta (t^{'}-t)+\theta (t-t^{'})\right)
  }{e^{\frac{\beta (2 {\bf{k.q}}-q^2)}{2 m}+\lambda + \lambda^{'} }-1}
\]

-e^{-\frac{i q^2 t^{'} }{2 m}}\mbox{         } G_{ {\bf{k}},> }(\lambda, \lambda^{'}; t,t^{'})  \mbox{             } n_F({\bf{k}}-{\bf{q}})
\mbox{               }
e^{\frac{i  (2 {\bf{k.q}} t+q^2 (t^{'}-t))}{2 m}} }{e^{\frac{\beta (2 {\bf{k.q}}-q^2)}{2 m}+\lambda + \lambda^{'} }-1}

\[
